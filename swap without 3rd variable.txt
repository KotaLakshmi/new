#include <stdio.h>
void swap(int *, int *);
void main()
{
    int a,b;
    printf("enter a and b value:");
    scanf("%d%d",&a,&b);
    printf("before swap:\t %d \t %d\n",a,b);
    swap(&a,&b);
    printf("after swap:\t %d \t %d",a,b);
}
void swap(int* x,int* y)
{
    
    *x=*x+*y;
    *y=*x-*y;
    *x=*x-*y;
    
}
